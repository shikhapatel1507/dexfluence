'use strict';

const http = require('http');
const fs = require('fs');
const path = require('path');

const PORT = parseInt(process.env.PORT || '5000', 10);

// Readiness flag
let isReady = false;

// Preload index.html if you want (optional, not used for health)
let cachedHtml = null;
try {
  cachedHtml = fs.readFileSync(
    path.join(__dirname, '..', 'dist', 'public', 'index.html'),
    'utf8'
  );
} catch (_) {}

// Create basic HTTP server
const server = http.createServer((req, res) => {
  // Health endpoints
  if (req.url === '/' || req.url === '/health' || req.url === '/healthz') {
    if (!isReady) {
      res.statusCode = 503;
      res.setHeader('Content-Type', 'application/json');
      res.end(JSON.stringify({ status: 'starting' }));
      return;
    }
    res.statusCode = 200;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ status: 'ok' }));
    return;
  }

  // Before ready, block other traffic
  if (!isReady) {
    res.statusCode = 503;
    res.setHeader('Content-Type', 'application/json');
    res.end(JSON.stringify({ error: 'Service warming up' }));
    return;
  }

  // Once ready, the real app (from dist/index.cjs) will have
  // replaced/augmented the server handlers internally.
  res.statusCode = 404;
  res.end('Not Found');
});

server.listen(PORT, '0.0.0.0', function () {
  const t = new Date().toLocaleTimeString();
  process.stdout.write(t + ' [start] serving on port ' + PORT + '\n');
});

// Expose server for the real app to hook into (your existing pattern)
global.__preloadServer = server;

// Load the real app asynchronously
setImmediate(function () {
  try {
    require(path.join(__dirname, '..', 'dist', 'index.cjs'));

    // If we reach here, the app loaded successfully
    isReady = true;
    process.stdout.write('[start] app loaded, service is ready\n');
  } catch (err) {
    process.stderr.write('[start] failed to load app: ' + (err && err.message) + '\n');
    // Stay not ready so Cloud Run never routes traffic
  }
});
